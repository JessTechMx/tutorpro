class Estudiante:
    def __init__(self, nombre):
        self.nombre = nombre
        self.progreso = 0

    def aprender(self, tema):
        self.progreso += 10
        print(f"¡{self.nombre}
        domina {tema}!")

alumno = Estudiante("Ana")
alumno.aprender("POO")
# ¡Ana domina POO!
